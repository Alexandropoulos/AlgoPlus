307 std::shared_ptr<node> _remove(std::shared_ptr<node> root, interval i) {
308 if (!root) {
309 return nullptr;
310 }
311 std::shared_ptr<node> p = root;
312 while (p) {
313 if (p->i->low > i.low) {
314 p = p->left;
315 } else if (p->i->low < i.low) {
316 p = p->right;
317 } else {
318 if (!p->right && !p->left) {
319 return nullptr;
320 } else if (p->right && !p->left) {
321 std::shared_ptr<node> temp = root->right;
322 return temp;
323 } else if (p->left && !p->right) {
324 std::shared_ptr<node> temp = p->left;
325 return temp;
326 } else {
327 std::shared_ptr<node> temp = root;
328 while (temp && temp->left) {
329 temp = temp->left;
330 }
331 p->i = temp->i;
332 p->max = temp->max;
333 p->right = _remove(p->right, *temp->i);
