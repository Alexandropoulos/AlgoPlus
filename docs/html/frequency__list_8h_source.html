215template<typename T>
216void frequency_list<T>::swap_nodes(std::shared_ptr<node> a, std::shared_ptr<node> b) {
217 if(a == nullptr || b == nullptr || a->next != b)
218 return;
219
220 // Attach b's next to a and a's prev to b
221 a->next = b->next;
222 b->prev = a->prev;
223
224 // If b's next exists, attach a to its prev
225 if(b->next)
226 b->next->prev = a;
227
228 // If a's prev exists(i.e., a is not the head), attach b to its next
229 if(a->prev.lock())
230 a->prev.lock()->next = b;
231
232 // Now, attach a to b's next and b to a's prev
233 b->next = a;
234 a->prev = b;
235
236 // If the node a was the head node, update the head to b
237 if (head == a)
238 head = b;
239}

298
299 // special case: deleting the head node

301 head = head->next;
302 if(head)
303 head->prev.reset();
