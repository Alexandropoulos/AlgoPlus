176 if (empty()) {
177 return;
178 }
179 std::shared_ptr<node> t = root;
180 std::shared_ptr<node> to_be_removed = nullptr;
181 while (t != tail && t->next->val != key) {
182 t = t->next;
183 }
184 if (t == tail) {
185 return;
186 }
187 to_be_removed = t->next;
188 t->next = t->next->next;
189 to_be_removed.reset();
190 if (t->next == nullptr) {
191 tail = t;

193 if (root == tail) {
194 tail = nullptr;
